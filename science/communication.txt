## 2 Rest API URIs

-   /api/utils
    -   get /info # info about this rest api server
    -   get /time # get time for timesync purposes

-   /api/auth
    authentication & authorization stuff
    -   post /begin
    -   post /complete
    -   post /forgotpass
    -   post /resetpass
    -   post /ping

-   /api/user
    user profile management
    -   post /create
    -   post /update
    -   get /whoami
    -   get /find
    -   get /in-project

-   /api/project
    project management
    -   post /create
    -   post /update
    -   post /clone
    -   get /find
    -   get /own-list
    -   delete /delete

-   /api/project/involvement
    project involvement management
    -   post /create
    -   post /update
    -   get /view?personId={personId}&projectId={projectId}
    -   get /list?page={page}&projectId={projectId}&personId={personId}
    -   delete /delete?personId={personId}&projectId={projectId}

-   /api/plugin
    plugin management
    -   post /create
    -   post /update
    -   get /view?id={id}
    -   get /list?page={page}&query={query}&authorId={authorId}
